% Complete recipe example
\begin{recipe}
[% 
    portion = {\portion{1-2}},
]
{Cucumber Salad}
    \ingredients{%
        1 & Cucumber\\
        & Salt\\
        & Garlic\\
        & Vinegar\\
        A little & Soy sauce
    }
    
    \preparation{%
        \step Smash the cucumber with a knife and cut into \unit[1-2]{cm} cubes.
        \step Mix cucumber with salt and wait for the water to be drawn out. Once the water is drawn out (around \unit[5]{min}), squeeze out the water.
        \step Mix in other ingredients and serve.
    }
    
\end{recipe}
