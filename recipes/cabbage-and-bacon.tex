% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[10]{min}},
    portion = {\portion{1-2}},
]
{Cabbage and Crispy Pork}
    \ingredients{%
        5-10 & Cabbage leaves\\
        3 & Bacon slices\\
        & Garlic\\
        & Bean paste\\
        & Soy sauce
    }
    
    \preparation{%
        \step Tear cabbage into \unit[1-2]{in} pieces.
        \step Cut bacon into small pieces.
        \step Gently fry bacon until crispy.
        \step Add garlic and a little bean paste.
        \step Add cabbage and garlic, fry until cooked.
        \step Add a little soy sauce.
    }
    
\end{recipe}
