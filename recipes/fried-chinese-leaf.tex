\begin{recipe}
[% 
    preparationtime = {\unit[5]{min}},
    portion = {\portion{1-2}},
]
{Fried Chinese Leaf}
    \ingredients{%
        6-8 & Chinese leaves\\
        & Soy sauce\\
        & Oyster sauce\\
        & Minced garlic
    }
    
    \preparation{%
        \step Cut leaves. The white parts should be cut into inch thick slices and the green parts should be cut thicker.
        \step Fry peppercorns and garlic until fragrant.
        \step Fry the white parts until tender.
        \step Add bean paste and mix in until coloured.
        \step Add the green parts and fry until cooked.
        \step Add soy sauce sauce, oyster sauce, and salt
    }
    
\end{recipe}
